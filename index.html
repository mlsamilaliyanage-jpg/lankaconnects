<script>

  function switchTab(n) {

    document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));

    document.getElementById('tab-' + n).classList.add('active');

    

    document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active', 'text-[#8D153A]', 'font-semibold'));

    document.querySelectorAll('.tab-btn')[n].classList.add('active', 'text-[#8D153A]', 'font-semibold');

  }


  // Sample Jobs

  const jobs = [

    {title:"Full Stack Developer", company:"Careem", salary:"18k-26k", location:"Dubai"},

    {title:"Senior Accountant", company:"Emirates NBD", salary:"12k-18k", location:"Abu Dhabi"},

    {title:"Heavy Driver", company:"Al Futtaim", salary:"5.5k+", location:"Sharjah"}

  ];


  function renderJobs() {

    const container = document.getElementById('jobs-grid');

    container.innerHTML = jobs.map(j => `

      <div class="bg-white p-6 rounded-3xl card-hover">

        <h3 class="font-semibold text-xl">${j.title}</h3>

        <p class="text-[#8D153A]">${j.company}</p>

        <p class="mt-3">${j.salary} • ${j.location}</p>

        <button onclick="alert('Applied successfully! 🎉')" class="mt-6 w-full bg-[#8D153A] text-white py-3 rounded-2xl">Apply Now</button>

      </div>

    `).join('');

  }


  function showPostModal() {

    document.getElementById('postModal').classList.remove('hidden');

  }


  function hidePostModal() {

    document.getElementById('postModal').classList.add('hidden');

  }


  // Initialize

  window.onload = () => {

    renderJobs();

    switchTab(0);

  };

</script>
